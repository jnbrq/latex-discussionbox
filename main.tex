\usepackage{discussionbox}
\renewcommand{\familydefault}{\sfdefault}

\DefinePerson{Canberk}{blue}
\DefinePerson{PersonA}{magenta}
\DefinePerson{PersonB}{blue}

\begin{document}

\begin{discussionbox}[Tutorial (D\dbox{nextid})]
    \Canberk{Welcome to the DiscussionBox tutorial!}
    \Canberk{
        DiscussionBox provides a forum-like typesetting for discussions taking place on \LaTeX{} document sharing platforms like Overleaf.
        Compared to review mechanisms provided by such platforms, DiscussionBox enables non-volatile and structured conversations, similar what e-mail.
        Unlike e-mail, however, you can benefit from powerful \LaTeX{} features!
    }
    \Canberk{
        DiscussionBox has two components: messages and a box to contain them.
    }
\end{discussionbox}

\begin{discussionbox}
    \Canberk{
        Hey, what's up?
        \PersonA{
            It's perfect! By the way, math formatting works:
            \begin{align}
                F = ma
            \end{align}
            \PersonA{Wow, that's cool!}
        }
        \PersonB{
            Same here! Check out the following Latex logo:

            \begin{center}
                \includegraphics[width=0.5\linewidth]{fig/latex-logo.png}
            \end{center}
            \PersonA{
                Wow, that's so good!
            }
            \Canberk{
                I totally agree!
                \Canberk{\LabelMsg{msg:canberk} More and more nesting...}
            }
            Previous message continues...
            \PersonA{Here is a long text:\\ \lipsum[1-2]}
        }
    }
\end{discussionbox}
